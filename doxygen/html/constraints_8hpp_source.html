51 fill_DBC(

54 const unsigned int dof_idx,
55 const bool set_inhomogeneities = true,
56 const bool close = true)
57 {

65 scratch_data.get_dof_handler(dof_idx),
66 bc.first,
67 *bc.second,
68 scratch_data.get_constraint(dof_idx));
69 else
70 dealii::DoFTools::make_zero_boundary_constraints<dim>(
71 scratch_data.get_dof_handler(dof_idx),
72 bc.first,
73 scratch_data.get_constraint(dof_idx));
74 }
75 }
76
77 if (close)
78 {
79 scratch_data.get_constraint(dof_idx).close();
80 check_constraints(scratch_data.get_dof_handler(dof_idx),
