641 Material<number>::Material(const MaterialData<number> &material_data,
642 const MaterialTypes material_type)
643 : data(material_data)
644 , gas(data.gas)
645 , liquid(data.liquid)
646 , solid(data.solid)
647 , material_type(material_type)
648 , inv_mushy_interval(material_type == MaterialTypes::liquid_solid ||
649 material_type == MaterialTypes::gas_liquid_solid ||

653 0.0)
