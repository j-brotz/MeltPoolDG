177 template <int dim, typename number>
178 dealii::Tensor<1, dim, dealii::VectorizedArray<number>>
179 normalize(const dealii::Tensor<1, dim, dealii::VectorizedArray<number>> &in,
180 const number zero = 1e-16)
181 {
182 dealii::Tensor<1, dim, dealii::VectorizedArray<number>> vec;
183
184 const auto n_norm = in.norm();
185
186 for (unsigned int v = 0; v < dealii::VectorizedArray<number>::size(); ++v)
187 if (n_norm[v] > zero)
188 for (unsigned int d = 0; d < dim; ++d)
189 vec[d][v] = in[d][v] / n_norm[v];
190 else
191 for (unsigned int d = 0; d < dim; ++d)
192 vec[d][v] = in[d][v] / zero;
193
194 return vec;
