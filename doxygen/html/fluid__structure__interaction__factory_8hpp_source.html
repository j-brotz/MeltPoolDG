62 fsi_obstacle_load = std::make_unique<ObstacleLoad<dim, number, ObstacleType>>(
63 StokesLawSphericalParticleForce<dim, number, ObstacleType>(
64 flow_solution, flow_mf_context, flow_material.dynamic_viscosity));
65 break;
66 }

70 }
