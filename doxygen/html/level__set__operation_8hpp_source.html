223 inline number
224 approximate_distance_from_level_set(const number phi, const number eps, const number cutoff)
225 {
226 if (std::abs(phi) < cutoff)
227 return eps * std::log((1. + phi) / (1. - phi));
228 else if (phi >= cutoff)
229 return eps * std::log((1. + cutoff) / (1. - cutoff));
230 else /*( phi <= -cutoff )*/
231 return -eps * std::log((1. + cutoff) / (1. - cutoff));
232 }
