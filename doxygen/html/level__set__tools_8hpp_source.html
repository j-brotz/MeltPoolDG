116 template <typename number>
117 inline number
118 approximate_distance_from_tanh_level_set(const number phi, const number eps, const number cutoff)
119 {
120 if (std::abs(phi) < cutoff)
121 return eps * std::log((1. + phi) / (1. - phi));
122 else if (phi >= cutoff)
123 return eps * std::log((1. + cutoff) / (1. - cutoff));
124 else /*( phi <= -cutoff )*/
125 return -eps * std::log((1. + cutoff) / (1. - cutoff));
126 }
