298 template <int dim, typename number>
299 void
300 ParticleIterator<dim, number>::make_accessor()
301 {
302 std::visit(
303 [&](auto &it) {
304 using T = std::decay_t<decltype(it)>;
305
306 if constexpr (std::is_same_v<T, dealii::Particles::ParticleIterator<dim>>)
307 particle_accessor.emplace(*it);
308 else if constexpr (std::is_same_v<T, PropertyPoolIteratorData>)
309 particle_accessor.emplace(*it.dealii_property_pool, it.handle);
310 else
311 AssertThrow(false, dealii::ExcInternalError());
312 },
313 deal_ii_iterator);
314 }
