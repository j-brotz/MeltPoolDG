100 {
101 dealii::TrilinosWrappers::SparseMatrix *preconditioner_matrix_ptr;

109 preconditioner_matrix_ptr = &preconditioner_system_matrix;

122 dealii::ExcMessage(
123 "No external system matrix has been passed to the preconditioner!"));
124 preconditioner_matrix_ptr =

126 }
