112 template <int dim>
113 inline double
114 compute_cell_diameter(const double cell_measure, const unsigned int fe_degree)
115 {
116 if constexpr (dim == 2)
117 return std::sqrt(4. * cell_measure / dealii::numbers::PI) / fe_degree;
118 else if constexpr (dim == 3)

121 AssertThrow(false, dealii::ExcMessage("Only dim=2 and dim=3 are supported."));

198 template <int dim>
199 double
200 find_point_triangle_distance(const std::vector<dealii::Point<dim>> &triangle,
201 const dealii::Point<dim> &point)
202 {
203 double distance = 0.0;
204
205 const dealii::Point<dim> &point_0 = triangle[0];
206 const dealii::Point<dim> &point_1 = triangle[1];
207
208 if constexpr (dim == 3)
209 {
210 const dealii::Point<dim> &point_2 = triangle[2];
211
212 const dealii::Tensor<1, dim> e_0 = point_1 - point_0;
213 const dealii::Tensor<1, dim> e_1 = point_2 - point_0;
214
215 const double a = e_0.norm_square();
216 const double b = scalar_product(e_0, e_1);
217 const double c = e_1.norm_square();
218 const double det = a * c - b * b;
219
220 const dealii::Tensor<1, dim> vector_to_plane = point_0 - point;
221
222 const double d = scalar_product(e_0, vector_to_plane);
223 const double e = scalar_product(e_1, vector_to_plane);
224
225 double s = b * e - c * d;
226 double t = b * d - a * e;
227
228 if (s + t <= det)
229 {
230 if (s < 0)
231 {
232 if (t < 0)
233 {
234 if (d < 0)
235 {
236 t = 0;
237 if (-d >= a)
238 s = 1;
239 else
240 s = -d / a;
241 }
242 else
243 {
244 s = 0;
245 if (e >= 0)
246 t = 0;
247 else if (-e >= c)
248 t = 1;
249 else
250 t = e / c;
251 }
252 }
253 else
254 {
255 s = 0;
256 if (e >= 0)
257 t = 0;
258 else if (-e >= c)
259 t = 1;
260 else
261 t = -e / c;
262 }
263 }
264 else if (t < 0)
265 {
266 t = 0;
267 if (d >= 0)
268 s = 0;
269 else if (-d >= a)
270 s = 1;
271 else
272 s = -d / a;
273 }
274 else
275 {
276 const double inv_det = 1. / det;
277 s *= inv_det;
278 t *= inv_det;
279 }
280 }
281 else
282 {
283 if (s < 0)
284 {
285 const double tmp0 = b + d;
286 const double tmp1 = c + e;
287 if (tmp1 > tmp0)
288 {
289 const double numer = tmp1 - tmp0;
290 const double denom = a - 2 * b + c;
291 if (numer >= denom)
292 s = 1;
293 else
294 s = numer / denom;
295 t = 1 - s;
296 }
297 else
298 {
299 s = 0;
300 if (tmp1 <= 0)
301 t = 1;
302 else if (e >= 0)
303 t = 0;
304 else
305 t = -e / c;
306 }
307 }
308 else if (t < 0)
309 {
310 const double tmp0 = b + e;
311 const double tmp1 = a + d;
312 if (tmp1 > tmp0)
313 {
314 const double numer = tmp1 - tmp0;
315 const double denom = a - 2 * b + c;
316 if (numer >= denom)
317 t = 1;
318 else
319 t = numer / denom;
320 s = 1 - t;
321 }
322 else
323 {
324 t = 0;
325 if (tmp1 <= 0)
326 s = 1;
327 else if (d >= 0)
328 s = 0;
329 else
330 s = -d / a;
331 }
332 }
333 else
334 {
335 const double numer = (c + e) - (b + d);
336 if (numer <= 0)
337 s = 0;
338 else
339 {
340 const double denom = a - 2 * b + c;
341 if (numer >= denom)
342 s = 1;
343 else
344 s = numer / denom;
345 }
346 t = 1 - s;
347 }
348 }
349
350 const dealii::Point<dim> pt_in_triangle = point_0 + s * e_0 + t * e_1;
351 distance = pt_in_triangle.distance(point);
352 }
353
354 if constexpr (dim == 2)
355 {
356 const dealii::Tensor<1, dim> d = point_1 - point_0;
357 const double t_bar = d * (point - point_0) / (d.norm() * d.norm() + 1e-12);
358
359 if (t_bar <= 0.0)
360 distance = (point - point_0).norm();
361 else if (t_bar >= 1.0)
362 distance = (point - point_1).norm();
363 else
364 distance = (point - (point_0 + t_bar * d)).norm();
365 }
366
367 return distance;
368 }
