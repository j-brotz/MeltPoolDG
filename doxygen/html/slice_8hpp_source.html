86 for (const auto &i : idx_req_vars)

90 // element wise ouput

93 std::get<3>(data),

95 std::get<4>(data));
96 // nodal output
97 else
98 data_out.add_data_vector(*std::get<0>(data),
99 *std::get<1>(data),
100 std::get<3>(data),
101 std::get<4>(data));
102 }

106 data_out.build_patches();

108 // write slice data to vtu files
109 const std::string pvtu_filename =

112 n_time_step,
113 tria_slice.get_mpi_communicator(),
114 output_data.paraview.n_digits_timestep,
115 output_data.paraview.n_groups);
116
117 // write a pvd file relating the pvtu-file with a simulation time
