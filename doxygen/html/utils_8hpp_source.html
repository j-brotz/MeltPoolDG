187 template <int dim, typename number>
188 inline DEAL_II_ALWAYS_INLINE //
189 dealii::Tensor<1, dim, dealii::VectorizedArray<number>>
190 calculate_velocity(const ConservedVariablesType<dim, number> &conserved_variables)
191 {
192 const dealii::VectorizedArray<number> inverse_density =
193 dealii::VectorizedArray<number>(1.) / conserved_variables[0];
194
195 dealii::Tensor<1, dim, dealii::VectorizedArray<number>> velocity;
196 for (unsigned int d = 0; d < dim; ++d)
197 velocity[d] = conserved_variables[1 + d] * inverse_density;
198
199 return velocity;
200 }

202 template <int dim, typename number>
203 inline DEAL_II_ALWAYS_INLINE //
204 dealii::Tensor<2, dim, dealii::VectorizedArray<number>>
205 calculate_grad_velocity(
206 const ConservedVariablesType<dim, number> &conserved_variables,
207 const ConservedVariablesGradientType<dim, number> &grad_conserved_variables)
208 {
209 const dealii::VectorizedArray<number> inverse_density =
210 dealii::VectorizedArray<number>(1.) / conserved_variables[0];
211 const dealii::Tensor<1, dim, dealii::VectorizedArray<number>> velocity =
212 calculate_velocity<dim, number>(conserved_variables);
213
214 dealii::Tensor<1, dim, dealii::VectorizedArray<number>> grad_rho;
215 for (unsigned int d = 0; d < dim; ++d)
216 grad_rho[d] = grad_conserved_variables[0][d];
217
218 dealii::Tensor<2, dim, dealii::VectorizedArray<number>> grad_rho_velocity;
219 for (unsigned int d = 0; d < dim; ++d)
220 for (unsigned int e = 0; e < dim; ++e)
221 grad_rho_velocity[d][e] = grad_conserved_variables[1 + d][e];
222
223 dealii::Tensor<2, dim, dealii::VectorizedArray<number>> grad_velocity;
224 for (unsigned int d = 0; d < dim; ++d)
225 for (unsigned int e = 0; e < dim; ++e)
226 grad_velocity[d][e] =
227 inverse_density * (grad_rho_velocity[d][e] - velocity[d] * grad_rho[e]);
228
229 return grad_velocity;
230 }

243 template <int dim, typename Number>
244 void
245 calculate_penalty_parameter(
246 dealii::AlignedVector<dealii::VectorizedArray<Number>> &array_penalty_parameter,
247 const dealii::MatrixFree<dim, Number> &matrix_free,
248 const std::string &domain_representation_type,
249 const unsigned int dof_index,
250 const Number scaling_factor)
251 {
252 const unsigned int n_cells = matrix_free.n_cell_batches() + matrix_free.n_ghost_cell_batches();
253 array_penalty_parameter.resize(n_cells);
254
255 dealii::Mapping<dim> const &mapping = *matrix_free.get_mapping_info().mapping;
256 dealii::FiniteElement<dim> const &fe = matrix_free.get_dof_handler(dof_index).get_fe();
257 unsigned int const degree = fe.degree;
258
259 // use penalty factor for hypercube elements according to K. Hillewaert, Development of the
260 // discontinuous Galerkin method for high-resolution, large scale CFD and acoustics in
261 // industrial geometries, PhD thesis, Univ. de Louvain, 2013.
262 const Number fac = scaling_factor * (degree + 1.0) * (degree + 1.0);
263
264 const std::vector<dealii::ReferenceCell<dim>> reference_cells =
265 matrix_free.get_dof_handler(dof_index).get_triangulation().get_reference_cells();
266 AssertThrow(reference_cells.size() == 1, dealii::ExcMessage("No mixed meshes allowed."));
267
268
269 const dealii::Quadrature<dim> quadrature =
270 reference_cells[0].get_gauss_type_quadrature(degree + 1);
271 dealii::FEValues<dim> fe_values(mapping, fe, quadrature, dealii::update_JxW_values);
272
273 const dealii::Quadrature<dim - 1> face_quadrature =
274 reference_cells[0].face_reference_cell(0).get_gauss_type_quadrature(degree + 1);
275 dealii::FEFaceValues<dim> fe_face_values(mapping,
276 fe,
277 face_quadrature,
278 dealii::update_JxW_values);
279
280 if (domain_representation_type == "fitted")
281 {
282 for (unsigned int i = 0; i < n_cells; ++i)
283 {
284 for (unsigned int v = 0; v < matrix_free.n_active_entries_per_cell_batch(i); ++v)
285 {
286 typename dealii::DoFHandler<dim>::cell_iterator cell =
287 matrix_free.get_cell_iterator(i, v, dof_index);
288 fe_values.reinit(cell);
289
290 // calculate cell volume
291 Number volume = 0;
292 for (unsigned int q = 0; q < quadrature.size(); ++q)
293 {
294 volume += fe_values.JxW(q);
295 }
296
297 // calculate surface area
298 Number surface_area = 0;
299 for (unsigned int const f : cell->face_indices())
300 {
301 fe_face_values.reinit(cell, f);
302 Number const factor =
303 (cell->at_boundary(f) and not(cell->has_periodic_neighbor(f))) ? 1. : 0.5;
304 for (unsigned int q = 0; q < face_quadrature.size(); ++q)
305 {
306 surface_area += fe_face_values.JxW(q) * factor;
307 }
308 }
309
310 array_penalty_parameter[i][v] = surface_area / volume * fac;
311 }
312 }
313 }
314 else if (domain_representation_type == "cut")
315 {
316 for (unsigned int i = 0; i < n_cells; ++i)
317 {
318 for (unsigned int v = 0; v < matrix_free.n_active_entries_per_cell_batch(i); ++v)
319 {
320 typename dealii::DoFHandler<dim>::cell_iterator cell =
321 matrix_free.get_cell_iterator(i, v, dof_index);
322
323 // simplified computation for hypercube elements
324 array_penalty_parameter[i][v] = fac / cell->minimum_vertex_distance();
325 }
326 }
327 }
328 else
329 AssertThrow(false,
330 dealii::ExcMessage("The domain representation type '" +
331 domain_representation_type + "' is not supported."));
332 }

334 template <int dim, typename number>
335 void
336 update_primitive_variables_solution(
337 dealii::LinearAlgebra::distributed::Vector<number> &solution_primitive_variables,
338 const dealii::LinearAlgebra::distributed::Vector<number> &solution,
339 const ScratchData<dim, dim, number> &scratch_data,
340 const unsigned int dof_idx,
341 const unsigned int quad_idx,
342 const Material<dim, number> *material_liquid,
343 const Material<dim, number> *material_gas)
344 {
345 const dealii::MatrixFree<dim, number, dealii::VectorizedArray<number>> &matrix_free =
346 scratch_data.get_matrix_free();
347 unsigned int n_support_points_per_cell = scratch_data.get_n_dofs_per_cell(dof_idx) / (dim + 2);
348
349 auto process_cell = [&](CutUtil::CellCategory category,
350 unsigned int first_component,
351 const Material<dim, number> *material,
352 const unsigned int cell_batch) {
353 if (!material)
354 return;
355 auto eval = FECellIntegrator<dim, dim + 2, number>(
356 matrix_free, dof_idx, quad_idx, first_component, category);
357 eval.reinit(cell_batch);
358 eval.read_dof_values(solution);
359
360 for (unsigned int i = 0; i < n_support_points_per_cell; ++i)
361 {
362 const auto &u_cons = eval.get_dof_value(i);
363 auto u_prim = material->eos_utils->convert_conservative_into_primitive_variables(u_cons);
364 eval.submit_dof_value(u_prim, i);
365 }
366
367 eval.set_dof_values(solution_primitive_variables);
368 };
369
370 if (!matrix_free.get_dof_info(dof_idx).cell_active_fe_index.empty())
371 {
372 // using hp::FECollection (cutFEM/DG)
373 for (unsigned int cell_batch = 0; cell_batch < matrix_free.n_cell_batches(); ++cell_batch)
374 {
375 const auto cell_category = matrix_free.get_cell_category(cell_batch);
376
377 if (cell_category == CutUtil::CellCategory::liquid)
378 {
379 process_cell(CutUtil::CellCategory::liquid, 0, material_liquid, cell_batch);
380 }
381 else if (cell_category == CutUtil::CellCategory::gas)
382 {
383 process_cell(CutUtil::CellCategory::gas, dim + 2, material_gas, cell_batch);
384 }
385 else if (cell_category == CutUtil::CellCategory::intersected)
386 {
387 process_cell(CutUtil::CellCategory::intersected, 0, material_liquid, cell_batch);
388 process_cell(CutUtil::CellCategory::intersected, dim + 2, material_gas, cell_batch);
389 }
390 }
391 }
392 else
393 {
394 // not using hp::FECollection (fitted mesh)
395 for (unsigned int cell_batch = 0; cell_batch < matrix_free.n_cell_batches(); ++cell_batch)
396 {
397 process_cell(CutUtil::CellCategory::liquid, 0, material_liquid, cell_batch);
398 }
399 }
400 }

402 template <typename DofViewType, typename VectorizedArrayType>
403 inline DEAL_II_ALWAYS_INLINE //
404 VectorizedArrayType
405 maximum_local_wave_speed(const DofViewType &u_m, const DofViewType &u_p)
406 {
407 const auto velocity_m = u_m.velocity();
408 const auto velocity_p = u_p.velocity();
409
410 const auto sound_speed_p = u_p.speed_of_sound();
411 const auto sound_speed_m = u_m.speed_of_sound();
412
413 const auto sound_speed_p2 = sound_speed_p * sound_speed_p;
414 const auto sound_speed_m2 = sound_speed_m * sound_speed_m;
415
416 const auto lambda = 0.5 * std::sqrt(std::max(velocity_p.norm_square() + sound_speed_p2,
417 velocity_m.norm_square() + sound_speed_m2));
418 return lambda;
419 }
