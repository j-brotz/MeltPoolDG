80 for (unsigned cell_batch = 0; cell_batch < mf_context.mf.n_cell_batches(); ++cell_batch)

82 FECellIntegrator<dim, n_components, number> phi(mf_context.mf,
83 mf_context.dof_idx,

85 phi.reinit(cell_batch);

87 phi.read_cell_data(cell_aligned_vector);
