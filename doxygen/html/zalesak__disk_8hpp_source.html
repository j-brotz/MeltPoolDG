82 void
83 set_boundary_conditions() override
84 {
85 /*
86 * create a pair of (boundary_id, dirichlet_function)
87 */
88 constexpr dealii::types::boundary_id inflow_bc = 42;
89 constexpr dealii::types::boundary_id do_nothing = 0;
90
91 this->attach_boundary_condition(
92 {inflow_bc, std::make_shared<dealii::Functions::ConstantFunction<dim>>(1)},
93 "dirichlet",
94 "level_set");
95 /*
96 * mark inflow edges with boundary label (no boundary on outflow edges must be prescribed
97 * due to the hyperbolic nature of the analyzed problem
98 *
99 out in
100 (-1,1) +---------------+ (1,1)
101 | : |
102 in | : | out
103 |_______________|
104 | : |
105 out | : | in
106 | : |
107 +---------------+
108 * (-1,-1) in out (1,-1)
109 */
110 if constexpr (dim == 2)
111 {
112 for (const auto &cell : this->triangulation->cell_iterators())
113 for (const auto &face : cell->face_iterators())
114 if ((face->at_boundary()))
115 {
116 const number half_line = (right_domain + left_domain) / 2;
117
118 if (face->center()[0] == left_domain && face->center()[1] > half_line)
119 face->set_boundary_id(inflow_bc);
120 else if (face->center()[0] == right_domain && face->center()[1] < half_line)
121 face->set_boundary_id(inflow_bc);
122 else if (face->center()[1] == right_domain && face->center()[0] > half_line)
123 face->set_boundary_id(inflow_bc);
124 else if (face->center()[1] == left_domain && face->center()[0] < half_line)
125 face->set_boundary_id(inflow_bc);
126 else
127 face->set_boundary_id(do_nothing);
128 }
129 }
130 else
131 {
132 (void)do_nothing; // suppress unused variable for 1D
133 }
134 }
